ADD THIS FUNCTION ABOVE router.post("/interview"):

async function storePayloadReadyIfPresent(reply: string, orgId: string)
{ if (!reply.includes("\[PAYLOAD_READY\]")) return;

try { const markerIndex = reply.indexOf("\[PAYLOAD_READY\]"); const
afterMarker = reply.slice(markerIndex +
"\[PAYLOAD_READY\]".length).trim();

    const fencedMatch =
      afterMarker.match(/```json\s*([\s\S]*?)\s*```/i);

    const rawJson =
      fencedMatch ? fencedMatch[1] : afterMarker;

    const firstBrace = rawJson.indexOf("{");
    const lastBrace = rawJson.lastIndexOf("}");

    if (
      firstBrace === -1 ||
      lastBrace === -1 ||
      lastBrace <= firstBrace
    ) {
      throw new Error("JSON not isolated");
    }

    const candidate =
      rawJson.slice(firstBrace, lastBrace + 1).trim();

    const parsed =
      JSON.parse(candidate);

    if (
      !parsed ||
      typeof parsed !== "object" ||
      typeof parsed.orgName !== "string"
    ) {
      throw new Error("Invalid payload");
    }

    await db.execute(sql`
      UPDATE organizations
      SET site_config =
        ${JSON.stringify(parsed)}::jsonb
      WHERE id = ${orgId}
    `);

    console.log(
      `[PAYLOAD_READY] stored config for ${orgId}`
    );

} catch (err) {

    console.warn(
      "[PAYLOAD_READY] parse failed",
      err
    );

} }

function requireServiceKey(req,res,next){

const provided = req.headers\["x-pillar-service-key"\];

const expected = process.env.PILLAR_SERVICE_KEY;

if(provided!==expected){
