```ts
async function storePayloadReadyIfPresent(reply: string, orgId: string) {
  if (!reply.includes("[PAYLOAD_READY]")) return;

  try {
    const marker = "[PAYLOAD_READY]";
    const markerIndex = reply.indexOf(marker);
    const afterMarker = reply.slice(markerIndex + marker.length).trim();

    // Prefer fenced json if present
    const fencedMatch = afterMarker.match(/```json\s*([\s\S]*?)\s*```/i);
    let candidate = fencedMatch ? fencedMatch[1].trim() : afterMarker;

    // If there is extra text before/after JSON, isolate the first full object
    const firstBrace = candidate.indexOf("{");
    const lastBrace = candidate.lastIndexOf("}");
    if (firstBrace === -1 || lastBrace === -1 || lastBrace <= firstBrace) {
      throw new Error("Could not isolate JSON object after [PAYLOAD_READY]");
    }

    candidate = candidate.slice(firstBrace, lastBrace + 1).trim();

    const parsed = JSON.parse(candidate) as Record<string, unknown>;

    if (!parsed || typeof parsed !== "object") {
      throw new Error("Parsed payload is not an object");
    }

    if (typeof parsed.orgName !== "string" || !parsed.orgName.trim()) {
      throw new Error("Payload missing required orgName");
    }

    await db.execute(sql`
      UPDATE organizations
      SET site_config = ${JSON.stringify(parsed)}::jsonb
      WHERE id = ${orgId}
    `);

    console.log(
      `[interview] [PAYLOAD_READY] stored site_config for org ${orgId} (orgName="${parsed.orgName}")`
    );
  } catch (err) {
    console.warn(`[interview] [PAYLOAD_READY] parse/store failed for org ${orgId}:`, err);
  }
}
```

```ts
function requirePillarServiceKey(req: Request, res: Response, next: NextFunction) {
  const expected = process.env.PILLAR_SERVICE_KEY;
  const provided = req.headers["x-pillar-service-key"];

  if (!expected) {
    return res.status(500).json({ ok: false, error: "PILLAR_SERVICE_KEY not configured" });
  }

  if (provided !== expected) {
    return res.status(401).json({ ok: false, error: "Unauthorized" });
  }

  next();
}
```

```ts
type PillarMethod = "POST" | "PATCH" | "PUT" | "DELETE";

export async function pillarRequest(path: string, method: PillarMethod, body?: unknown) {
  const baseUrl = process.env.COMMUNITY_PLATFORM_URL;
  const serviceKey = process.env.PILLAR_SERVICE_KEY;

  if (!baseUrl) {
    throw new Error("COMMUNITY_PLATFORM_URL is not set");
  }

  if (!serviceKey) {
    throw new Error("PILLAR_SERVICE_KEY is not set");
  }

  const response = await fetch(`${baseUrl}${path}`, {
    method,
    headers: {
      "Content-Type": "application/json",
      "x-pillar-service-key": serviceKey,
    },
    body: body ? JSON.stringify(body) : undefined,
  });

  const text = await response.text();
  let data: any = null;

  try {
    data = text ? JSON.parse(text) : null;
  } catch {
    data = { raw: text };
  }

  if (!response.ok) {
    throw new Error(
      `Pillar sync failed ${method} ${path}: ${response.status} ${JSON.stringify(data)}`
    );
  }

  return data;
}
```
