Explicit instructions for Pillar to test the full interview → live site flow. Zero changes to Discover Irwin.

Step 1: Get the service key.

The Discover Irwin production site already has a PILLAR_SERVICE_KEY environment variable set. You (the Pillar developer) need to know this value. You set it during a previous session. If you don't remember it, ask the site owner for it. Do not guess.

Step 2: Store the URL and key in Pillar's database.

In Pillar's database, find the org record you're using for testing. Update it with these two values:

UPDATE organizations
SET community_site_url = 'https://discoverirwin.com',
    community_site_key = '[the PILLAR_SERVICE_KEY value from Step 1]'
WHERE id = [your test org id];
If your organizations table doesn't have these columns yet, add them:

ALTER TABLE organizations
ADD COLUMN IF NOT EXISTS community_site_url TEXT,
ADD COLUMN IF NOT EXISTS community_site_key TEXT;
Then run the UPDATE.

Step 3: Update the provision endpoint.

Your current provision endpoint saves to the database and returns ok: true without actually calling the deployed site. Replace it with this:

app.post("/api/community-site/provision", async (req, res) => {
  try {
    const org = await getAuthenticatedOrg(req);
    if (!org) {
      return res.status(401).json({ error: "Unauthorized" });
    }
    const siteUrl = org.community_site_url;
    const siteKey = org.community_site_key;
    if (!siteUrl) {
      return res.status(400).json({ 
        error: "No community site URL configured for this organization." 
      });
    }
    if (!siteKey) {
      return res.status(400).json({ 
        error: "No service key configured for this organization." 
      });
    }
    const payload = buildPayloadFromInterviewAnswers(org);
    const controller = new AbortController();
    const timeout = setTimeout(() => controller.abort(), 30000);
    let response;
    try {
      response = await fetch(siteUrl.replace(/\/$/, "") + "/api/pillar/setup", {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          "X-Pillar-Key": siteKey,
        },
        body: JSON.stringify(payload),
        signal: controller.signal,
      });
    } catch (fetchError) {
      clearTimeout(timeout);
      if (fetchError.name === "AbortError") {
        return res.status(504).json({ 
          error: "Request timed out. The site may still be starting up. Try again in a minute." 
        });
      }
      return res.status(502).json({ 
        error: "Could not reach the community site. Check the URL and try again." 
      });
    }
    clearTimeout(timeout);
    if (response.status === 401) {
      return res.status(502).json({ 
        error: "Service key rejected by the community site. The key in Pillar's database does not match the key on the deployed site." 
      });
    }
    if (response.status === 400) {
      const body = await response.text().catch(() => "");
      return res.status(400).json({ 
        error: "The community site rejected the payload. Check the payload format.",
        details: body
      });
    }
    if (!response.ok) {
      const body = await response.text().catch(() => "");
      return res.status(502).json({ 
        error: `Community site returned status ${response.status}.`,
        details: body
      });
    }
    const result = await response.json();
    if (!result.ok) {
      return res.status(502).json({ 
        error: "Community site did not confirm setup.",
        details: result
      });
    }
    return res.json({
      ok: true,
      siteUrl: siteUrl,
      message: "Your site is live!",
    });
  } catch (err) {
    console.error("Provision error:", err);
    return res.status(500).json({ 
      error: "Internal error during provisioning." 
    });
  }
});
Step 4: Build the payload from interview answers.

Add this function. It takes the stored interview answers and builds the exact JSON payload the setup API expects:

function buildPayloadFromInterviewAnswers(org) {
  const answers = org.interview_answers; // however you store them
  const orgName = answers.org_name || org.name || "My Organization";
  const shortName = answers.short_name || generateInitials(orgName);
  const orgType = answers.org_type || "Other";
  const tagline = answers.tagline || null;
  const city = answers.city || "";
  const state = answers.state || "";
  const stateAbbrev = abbreviateState(state);
  const location = city && stateAbbrev ? `${city}, ${stateAbbrev}` : city || stateAbbrev || null;
  const colors = getColorsForOrgType(orgType);
  const businessOrgs = [
    "Main Street / Downtown Association",
    "Chamber of Commerce", 
    "Neighborhood Association"
  ];
  const isBusiness = businessOrgs.includes(orgType);
  const thirdLabel = isBusiness ? "Local Businesses" : "Active Members";
  const partners = parsePartners(answers.partners || "");
  const meeting = parseMeetingSchedule(answers.meeting_schedule || "");
  const mission = tagline
    ? `${orgName} is dedicated to ${tagline.toLowerCase().replace(/\.$/, "")}. Bringing together our community through events, programs, and service${location ? " in " + location : ""}.`
    : `${orgName} serves our community${location ? " in " + location : ""} through events, programs, and service.`;
  const aboutMission = [
    `${orgName}${tagline ? " is dedicated to " + tagline.toLowerCase().replace(/\.$/, "") : " serves our community"}. ${location ? "Based in " + location + ", we" : "We"} bring together members, businesses, and residents through community events and programs.`,
    `We host ${answers.annual_events || "multiple"} events annually, bringing together ${answers.annual_attendees || "hundreds of"} attendees. ${isBusiness ? "Our area includes " + (answers.members_or_businesses || "many") + " local businesses." : "Our organization has " + (answers.members_or_businesses || "many") + " active members."}`,
    "We are a 100% volunteer-run organization. Our events and programs are made possible by the dedication of our members and the support of our community partners and sponsors."
  ].join("\n\n");
  return {
    orgName,
    shortName,
    orgType: orgType.toLowerCase(),
    tagline,
    mission,
    location,
    primaryColor: colors.primary,
    accentColor: colors.accent,
    logoUrl: null,
    heroImageUrl: null,
    contactEmail: answers.email || null,
    contactPhone: answers.phone || null,
    contactAddress: answers.address || null,
    mailingAddress: answers.mailing_address || answers.address || null,
    website: answers.website || null,
    socialFacebook: answers.social_facebook || null,
    socialInstagram: answers.social_instagram || null,
    socialTwitter: null,
    socialLinkedin: null,
    meetingDay: meeting.day,
    meetingTime: meeting.time,
    meetingLocation: meeting.location,
    footerText: `${orgName} is a volunteer organization serving the ${location || "local"} community.`,
    metaDescription: tagline
      ? `${orgName} — ${tagline}. Community events, programs, and more${location ? " in " + location : ""}.`
      : `${orgName}. Community events, programs, and more${location ? " in " + location : ""}.`,
    stats: [
      { value: answers.annual_events || "10+", label: "Annual Events" },
      { value: answers.annual_attendees || "500+", label: "Annual Attendees" },
      { value: answers.members_or_businesses || "50+", label: thirdLabel },
      { value: "100%", label: "Volunteer Run" },
    ],
    programs: [],
    partners,
    sponsorshipLevels: [],
    events: [],
    sponsors: [],
    businesses: [],
    siteContent: {
      home_tagline: tagline || "Welcome to " + shortName,
      home_intro: mission,
      home_subtitle: orgType === "Other" ? "Community Organization" : orgType,
      contact_address: answers.address || "",
      contact_phone: answers.phone || "",
      contact_email: answers.email || "",
      meeting_contact_email: answers.events_email || answers.email || "",
      social_facebook: answers.social_facebook || "",
      social_instagram: answers.social_instagram || "",
      about_mission: aboutMission,
      community_partners: JSON.stringify(partners),
    },
  };
}
Step 5: Add the helper functions.

function generateInitials(name) {
  if (!name) return "ORG";
  const skip = ["the", "of", "and", "&", "a", "an", "in", "for"];
  const words = name.split(/\s+/).filter(w => w.length > 0);
  if (words.length === 0) return "ORG";
  if (words.length === 1) return words[0].substring(0, 3).toUpperCase();
  return words
    .filter(w => !skip.includes(w.toLowerCase()))
    .map(w => w[0])
    .join("")
    .toUpperCase()
    .substring(0, 4);
}
function abbreviateState(state) {
  const map = {
    alabama: "AL", alaska: "AK", arizona: "AZ", arkansas: "AR", california: "CA",
    colorado: "CO", connecticut: "CT", delaware: "DE", florida: "FL", georgia: "GA",
    hawaii: "HI", idaho: "ID", illinois: "IL", indiana: "IN", iowa: "IA",
    kansas: "KS", kentucky: "KY", louisiana: "LA", maine: "ME", maryland: "MD",
    massachusetts: "MA", michigan: "MI", minnesota: "MN", mississippi: "MS",
    missouri: "MO", montana: "MT", nebraska: "NE", nevada: "NV",
    "new hampshire": "NH", "new jersey": "NJ", "new mexico": "NM", "new york": "NY",
    "north carolina": "NC", "north dakota": "ND", ohio: "OH", oklahoma: "OK",
    oregon: "OR", pennsylvania: "PA", "rhode island": "RI", "south carolina": "SC",
    "south dakota": "SD", tennessee: "TN", texas: "TX", utah: "UT", vermont: "VT",
    virginia: "VA", washington: "WA", "west virginia": "WV", wisconsin: "WI",
    wyoming: "WY", "district of columbia": "DC",
  };
  const lower = state.trim().toLowerCase();
  if (map[lower]) return map[lower];
  if (state.trim().length === 2) return state.trim().toUpperCase();
  return state.trim().substring(0, 2).toUpperCase();
}
function getColorsForOrgType(orgType) {
  const colors = {
    "Main Street / Downtown Association": { primary: "#c25038", accent: "#2b7ab5" },
    "Chamber of Commerce":               { primary: "#1a4a8a", accent: "#d4a017" },
    "Rotary Club":                        { primary: "#003DA5", accent: "#d4a017" },
    "Lions Club":                         { primary: "#d4a017", accent: "#1a4a8a" },
    "VFW / American Legion":             { primary: "#8b1a1a", accent: "#2b5797" },
    "Fraternal Organization":             { primary: "#1a3a5c", accent: "#c5a030" },
    "PTA / PTO":                          { primary: "#339966", accent: "#7a3d9e" },
    "Community Foundation":               { primary: "#2d8a57", accent: "#2b7ab5" },
    "Neighborhood Association":           { primary: "#c26a17", accent: "#338899" },
    "Arts Council":                       { primary: "#7a3d9e", accent: "#cc3366" },
  };
  return colors[orgType] || { primary: "#2b7ab5", accent: "#338899" };
}
function parsePartners(input) {
  if (!input || input.trim().length === 0) return [];
  const lines = input.split(/[;\n]/).map(l => l.trim()).filter(l => l.length > 0);
  return lines.map(line => {
    const separators = [" - ", " — ", " – ", ": ", " | "];
    for (const sep of separators) {
      const idx = line.indexOf(sep);
      if (idx > 0) {
        return { name: line.substring(0, idx).trim(), description: line.substring(idx + sep.length).trim() };
      }
    }
    return { name: line, description: "Community partner" };
  });
}
function parseMeetingSchedule(input) {
  if (!input || input.trim().length === 0) return { day: null, time: null, location: null };
  const text = input.trim();
  let day = null, time = null, location = null;
  const dayMatch = text.match(/every\s+(monday|tuesday|wednesday|thursday|friday|saturday|sunday)/i)
    || text.match(/(1st|2nd|3rd|4th|first|second|third|fourth)\s+(monday|tuesday|wednesday|thursday|friday|saturday|sunday)/i);
  if (dayMatch) day = dayMatch[0].trim();
  const timeMatch = text.match(/(\d{1,2}:\d{2}\s*(?:AM|PM|am|pm))/);
  if (timeMatch) time = timeMatch[1].trim();
  const parts = text.split(",").map(p => p.trim());
  if (parts.length >= 3) location = parts.slice(2).join(", ").trim();
  else if (parts.length === 2 && !parts[1].match(/\d{1,2}:\d{2}/)) location = parts[1];
  if (!day && !time && !location) return { day: text, time: null, location: null };
  return { day, time, location };
}
Step 6: Test it.

Complete the interview for a test org
Make sure that org has community_site_url = https://discoverirwin.com and community_site_key = the correct key in Pillar's database
Click "Launch Site"
The provision endpoint reads the URL and key, builds the payload, calls POST https://discoverirwin.com/api/pillar/setup, and returns success
You should see {"ok": true} come back and the org config on DI will update
WARNING: This will overwrite Discover Irwin's org config and site content with the test org's data. Only do this if the site owner is okay with it. The site content (events, businesses, sponsors) will NOT be deleted — only the org config and site_content key/value rows get overwritten. To restore DI's original config afterward, call the setup API again with DI's real data.
